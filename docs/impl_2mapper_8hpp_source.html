115  template<typename P, typename MT>
116  JSONMapper<T>&
117  JSONMapper<T>::registerField(const std::string& key, P T::* value) {
118  _writers.push_back(std::function<bool(const T&, std::ostream&)>([=, this](const T& obj, std::ostream& out) -> bool {
119  if constexpr (std::is_same<MT, P>::value) { // Mapper type has not been specified
120 
121  if constexpr (std::is_pointer<P>::value) { // Remove pointer on parameter type to check the main type
122  if constexpr (isSame<T, std::remove_pointer<P>()>()) { // if same, use direct call to avoid recursive instanciation
123  return this->serialize(obj.*value, out, std::optional<std::string>(key));
124  } else { // Use 'unpointered' type
125  return callMappingFunctionPtr(obj.*value, out, key);
126  }
127 
128  } else if constexpr (std::is_reference<P>::value) { // Remove reference on parameter type to check the main type
129  if constexpr (isSame<std::remove_reference<P>(), T>()) { // if same, use direct call to avoid recursive instanciation
130  return this->serialize(obj.*value, out, std::optional<std::string>(key));
131  } else { // Use 'unreferenced' type
132  return callMappingFunctionRef(obj.*value, out, key);
133  }
134 
135  } else { // Basic type
136  if constexpr (std::is_same<MT, T>::value) { // if same, use direct call to avoid recursive instanciation
137  return this->serialize(obj.*value, out, std::optional<std::string>(key));
138  } else {
139  return JSONFactory<MT>::getInstance().serialize(obj.*value, out, std::optional<std::string>(key));
140  }
141  }
142 
143  } else { // Parameter type has to be correctly set by caller when specified
144  if constexpr (std::is_same<MT, T>::value) { // if same, use direct call to avoid recursive instanciation
145  return this->serialize(obj.*value, out, std::optional<std::string>(key));
146  } else {
147  return JSONFactory<MT>::getInstance().serialize(obj.*value, out, std::optional<std::string>(key));
148  }
149  }
150  }));
151  return *this;
152  }

194  template<typename T>
195  bool
196  JSONMapper<T>::serialize(const T& value, std::ostream& out, const std::optional<std::string>& field) {
197  JSONMapper<T>& mapper = JSONFactory<T>::getInstance();
198  if(field.has_value()) {
199  out << STRING_DELIMITER << field.value() << STRING_DELIMITER << FIELD_VALUE_SEPARATOR;
200  }
201  out << OBJECT_BEGIN;
202  auto iter = mapper.getWritterMapping().begin();
203  while(true) {
204  bool added = std::invoke(*iter, value, out);
205  if(++iter == mapper.getWritterMapping().end()) {
206  break;
207  }
208  if(added) {
209  out << FIELD_SEPARATOR;
210  }
211  }
212  out << OBJECT_END;
213  return true;
214  }
