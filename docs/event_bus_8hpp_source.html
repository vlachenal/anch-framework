209 EventBus<Evt>::scheduleDeferred(const Evt& event) noexcept {
210 std::lock_guard<std::mutex> lock(_queueMutex);
211 bool empty = _events.empty();
212 _events.push(event);
213 if(empty) {
214 if(_thread != NULL) {
