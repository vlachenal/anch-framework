146 }

220 void initializeWriters(std::map<std::string,anch::logger::Writer*>& writers,
221 const anch::resource::Resource& resource) {
222 std::map<std::string,unsigned int> sizeMap = std::map<std::string,unsigned int>({
223 {"K",1024},
224 {"M",1024*1024},
225 {"G",1024*1024*1024}
226 });
227 const std::regex sizeRegex = std::regex("^([0-9]+)(K|M|G)?$");
228
229 // Writer QoS configuration +
230 bool globalThreadSafe = true;
231 std::string paramValue;
232 if(resource.getParameter(paramValue, "thread.safe")) {
233 globalThreadSafe = (paramValue == "1");
234 }
235 paramValue = "";
236 bool globalLowPriority = false;
237 if(resource.getParameter(paramValue, "low.priority")) {
238 globalLowPriority = (paramValue == "1");
239 }
240 paramValue = "";
241 // Writer QoS configuration -
242
243 std::smatch match;
244 const std::map<std::string,anch::resource::Section>& config = resource.getConfiguration();
245 for(auto iter = config.cbegin() ; iter != config.cend() ; iter++) {
246 if(iter->first.substr(0,8) == "WRITER::") {
247 std::string writer = iter->first;
248 std::string name = writer.substr(8);
249 // Retrieve log pattern +
250 std::string pattern;
251 bool found = resource.getParameter(pattern,
252 "writer.pattern",
253 writer);
254 if(!found) {
255 pattern = "%m";
256 }
257 // Retrieve log pattern -
258
259 // Writer QoS configuration +
260 bool threadSafe = globalThreadSafe;
261 if(resource.getParameter(paramValue, "writer.thread.safe", writer)) {
262 threadSafe = (paramValue == "1");
263 }
264 paramValue = "";
265 bool lowPriority = globalLowPriority;
266 if(resource.getParameter(paramValue, "writer.low.priority", writer)) {
267 lowPriority = (paramValue == "1");
268 }
269 // Writer QoS configuration -
270
271 if(name == "console") { // Console writer will put logs on standard output
272 writers[name] = createWriterInstance(threadSafe,
273 lowPriority,
274 (std::ostream*)(&std::cout),
275 pattern);
276
277 } else { // File writer
278 // File path +
279 std::string path;
280 found = resource.getParameter(path, "writer.filepath", writer);
281 // File path -
282 if(found) {
283 // Max file size +
284 std::string maxSizeStr;
285 unsigned int maxSize = 0;
286 resource.getParameter(maxSizeStr, "writer.max.size", writer);
287 if(regex_search(maxSizeStr, match, sizeRegex)) {
288 const std::string multStr = std::string(match[2].first,
289 match[2].second);
290 maxSize = static_cast<unsigned int>(std::stoi(std::string(match[1].first,
291 match[1].second).data()));
292 if(multStr != "") {
293 maxSize *= sizeMap[multStr];
294 }
295 }
296 // Max file size -
297
298 // Max file index on rotate +
299 int maxIndex = 0;
300 std::string maxIdxStr;
301 resource.getParameter(maxIdxStr,
302 "writer.max.rotate.index",
303 writer);
304 if(found) {
305 maxIndex = std::stoi(maxIdxStr);
306 }
307 // Max file index on rotate -
308 writers[name] = createWriterInstance(threadSafe,
309 lowPriority,
310 path,
311 pattern,
312 maxSize,
313 maxIndex);
314 }
315 }
316 }

326 void initializeLoggersConfiguration(const std::map<std::string,anch::logger::Writer*>& writers,
327 const anch::resource::Resource& resource) {
328 std::regex upperRegex = std::regex("[a-z]");
329 std::smatch match;
330 std::string upperRep = "[A-Z]";
331
332 const std::map<std::string,anch::resource::Section>& config = resource.getConfiguration();
333 for(auto iter = config.cbegin() ; iter != config.cend() ; iter++) {
334 if(iter->first.substr(0,10) == "CATEGORY::") {
335 std::string category = iter->first;
336 std::string name = iter->first.substr(10);
337 std::string writerStr;
338 bool found = resource.getParameter(writerStr,
339 "logger.writers",
340 category);
341 if(found) {
342 // Retrieve writers +
343 size_t pos = 0;
344 std::vector<anch::logger::Writer*> loggerWriters;
345 size_t nextPos = writerStr.find(',',pos);
346 while(nextPos != std::string::npos && nextPos != pos) {
347 const std::string writerName = writerStr.substr(pos,
348 nextPos - pos);
349 auto iter = writers.find(writerName);
350 if(iter != writers.cend()) {
351 loggerWriters.push_back(iter->second);
352 }
353 pos = nextPos + 1;
354 nextPos = writerStr.find(',',pos);
355 }
356 const std::string writerName = writerStr.substr(pos);
357 auto iter = writers.find(writerName);
358 if(iter != writers.cend()) {
359 loggerWriters.push_back(iter->second);
360 }
361 // Retrieve writers -
362
363 // Retrieve level +
364 if(loggerWriters.empty()) {
365 std::cerr << "No writers has been found for " << name << ": "
366 << writerStr << std::endl;
367
368 } else {
369 std::string levelLbl;
370 resource.getParameter(levelLbl, "logger.level", category);
371 std::transform(levelLbl.begin(),
372 levelLbl.end(),
373 levelLbl.begin(),
374 ::toupper);
375 anch::logger::Level level = anch::logger::Level::FATAL;
376 auto iterLvl = anch::logger::LABEL_LEVEL.find(levelLbl);
377 if(iterLvl == anch::logger::LABEL_LEVEL.cend()) {
378 std::cerr << "Invalid level: " << levelLbl << std::endl;
379 } else {
380 level = iterLvl->second;
381 }
382 _loggersConfig.push_back(LoggerConfiguration(name,
383 level,
384 loggerWriters));
385 }
386 // Retrieve level -
387 }
