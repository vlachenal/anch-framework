206  void expandKey(const uint8_t key[4 * K]) {
207  std::memcpy(_expKey.swKey, key, 4 * K);
208  for(std::size_t i = K ; i < 4 * (R + 1) ; ++i) {
209  uint32_t mod = static_cast<uint32_t>(i % K);
210  if(mod == 0) {
211  _expKey.swKey[i] = _expKey.swKey[i - K] ^ (subWord(rotateWord(_expKey.swKey[i - 1])) ^ ANCH_AES_RCON[i / K]);
212  } else if(K > 6 && mod == 4) {
213  _expKey.swKey[i] = _expKey.swKey[i - K] ^ subWord(_expKey.swKey[i - 1]);
214  } else {
215  _expKey.swKey[i] = _expKey.swKey[i - K] ^ _expKey.swKey[i - 1];
216  }
217  }

233  resBytes[1] = ANCH_AES_CIPHER_SBOX[bytes[1]];

235  resBytes[3] = ANCH_AES_CIPHER_SBOX[bytes[3]];
236  return res;
237  }
