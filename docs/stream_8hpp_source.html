354 Stream<T,C>::allMatch(std::function<bool(const T&)> predicate, const std::vector<std::function<bool(const T&)>>& filters) {
355 bool match = true;
356 for(auto val : _values) {
357 for(auto filter : filters) {
358 if(!filter(val)) {
359 goto next;
360 }
361 }

363 match = false;
364 break;
365 }
366 next:

368 break;
369 }
370 }
371 if(_next != NULL && match && !limitReached()) {
372 match = _next->allMatch(predicate, filters);
373 }
374 return match;
