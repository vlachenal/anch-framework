353 Stream<T,C>::allMatch(std::function<bool(const T&)> predicate, const std::vector<std::function<bool(const T&)>>& filters) {
354 bool match = true;
355 for(auto val : _values) {
356 for(auto filter : filters) {

358 goto next;
359 }

367 break;
368 }
369 }
370 if(_next != NULL && match && !limitReached()) {
371 match = _next->allMatch(predicate, filters);
372 }
373 return match;
