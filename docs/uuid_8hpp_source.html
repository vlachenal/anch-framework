211  auto epoch = now.time_since_epoch();

213  getRandomEngine().seed(timestamp);

219  case Uuid::Version::MAC_ADDRESS:

247  static void generateUuidVersion1(Uuid& uuid,
248  const std::string& macAddress = "") {
249  // Timestamp +
250  uint64_t timestamp = getUtcTimestamp();
251  uuid._lowTime = timestamp & TIME_LOW_MASK;
252  uuid._midTime = (timestamp >> 32) & TIME_MID_MASK;
253  uuid._highTime = (timestamp >> 48) & TIME_HIGH_MASK;
254  // Timestamp -
255 
256  // Sequence +
257  uint16_t clockSeq = getDistSeq()(getRandomEngine());
258  uuid._clockSeqLow = clockSeq & SEQ_LOW_MASK;
259  uuid._clockSeqHighRes = (clockSeq & SEQ_HIGH_MASK) >> 8;
260  uuid._clockSeqHighRes |= 0x80;
261  // Sequence -
262 
263  // MAC address +
264  std::string macAddr = macAddress;
265  if(macAddr == "") {
266  for(auto iter : anch::device::Network::getInterfaces()) {
267  const anch::device::NetworkInterface iface = iter.second;
268  if(!iface.isLocalhost()) {
269  macAddr = iface.getMacAddress();
270  }
271  }
272  }
273  std::size_t pos = 0;
274  std::size_t last = 0;
275  std::string macHexStr = "";
276  do {
277  if(pos != 0) {
278  last = pos + 1;
279  }
280  pos = macAddr.find(':', last);
281  macHexStr += macAddr.substr(last, pos - last);
282  } while(pos != std::string::npos);
283  uuid._node = std::stoull(macHexStr,0,16);
284  // MAC address -
285  }

293  static void generateUuidVersion3(Uuid& uuid, const std::string& data) {
294  anch::crypto::MD5 hash(data);
295  const std::array<uint8_t,16>& digest = hash.digest();
296 
297  // Timestamp +
298  uuid._lowTime = digest[0]
299  + (static_cast<uint32_t>(digest[1]) << 8)
300  + (static_cast<uint32_t>(digest[2]) << 16)
301  + (static_cast<uint32_t>(digest[3]) << 24);
302  uuid._midTime = digest[4]
303  + (static_cast<uint16_t>(digest[5]) << 8);
304  uuid._highTime = digest[6]
305  + (static_cast<uint16_t>(digest[7]) << 8);
306  uuid._highTime = uuid._highTime & TIME_HIGH_MASK;
307  // Timestamp -
308 
309  // Sequence +

311  uuid._clockSeqHighRes = digest[8];
312  uuid._clockSeqHighRes |= 0x80;
313  // Sequence -
314 
315  // Node +
316  uuid._node = digest[10]
317  + (static_cast<uint64_t>(digest[11]) << 8)
318  + (static_cast<uint64_t>(digest[12]) << 16)
319  + (static_cast<uint64_t>(digest[13]) << 24)
320  + (static_cast<uint64_t>(digest[14]) << 32)
321  + (static_cast<uint64_t>(digest[15]) << 40);

323  }
